CREATE DATABASE DB_BRECELL;
USE DB_BRECELL;
CREATE TABLE CELULAR
(
	COD_CELULAR INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    DES_CELULAR VARCHAR(50) NOT NULL,
    VAL_PRECO FLOAT NOT NULL,
    CAM_IMAGEM VARCHAR(200) NOT NULL,
    COD_MARCA INT NOT NULL,
    
    FOREIGN KEY FK_CELULAR_MARCA (COD_MARCA) REFERENCES MARCA (COD_MARCA)
);

CREATE TABLE MARCA 
(
	COD_MARCA INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
